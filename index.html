// --- INTELIGÊNCIA ARTIFICIAL ATUALIZADA COM RETRY --- document.getElementById('cameraInput').addEventListener('change', async (e) => { const file = e.target.files[0]; if (!file) return; document.getElementById('status').innerText = "Analisando com IA..."; const reader = new FileReader(); reader.onloadend = async () => { try { const base64Image = reader.result.split(',')[1]; const response = await fetch(API_URL, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ inputs: base64Image }) }); if (!response.ok) throw new Error("Erro no servidor backend"); const result = await response.json(); console.log("Resposta recebida da IA:", result); // CORREÇÃO: Fechamento correto da lógica e das chaves if (Array.isArray(result) && result[0].generated_text) { const descricao = result[0].generated_text; document.getElementById('p-nome').value = descricao; document.getElementById('form-produto').style.display = 'block'; document.getElementById('status').innerText = "Identificado!"; } else { document.getElementById('status').innerText = "Erro no formato da IA."; } } catch (err) { // A chave do try fecha aqui, permitindo que o catch funcione console.error(err); alert("Erro ao conectar com seu servidor backend."); document.getElementById('status').innerText = "Erro."; } }; reader.readAsDataURL(file); });
