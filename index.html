document.getElementById('cameraInput').addEventListener('change', async (e) => { const status = document.getElementById('status'); const file = e.target.files[0]; status.innerText = "Processando imagem..."; // Cria um elemento de imagem temporário para processamento const img = new Image(); img.src = URL.createObjectURL(file); img.onload = async () => { const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d'); // Define o tamanho para o canvas (podemos focar no centro da imagem) canvas.width = img.width; canvas.height = img.height; // Aplica filtros para destacar o texto ctx.filter = 'grayscale(1) contrast(1.5)'; ctx.drawImage(img, 0, 0); // Executa o OCR no canvas processado const { data: { text } } = await Tesseract.recognize(canvas, 'por', { logger: m => console.log(m) // Opcional: ver progresso no console }); document.getElementById('p-nome').value = text.split('\n')[0].trim(); status.innerText = "Leitura concluída!"; }; });
